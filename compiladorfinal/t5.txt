program tres
{
   var x : int;
   var y : int;
   var z: int
}
{
   read x;
   read y;

   if (y < x)
   {
     set z x;
     set x y;
     set y z;
   }

   set z 0;
   while (x < y) do
   {
     set z z + y;
     set x x + 1;
   }

   ifelse (x < 2*y + 2)
   {
     set x x + y;
     set y 0;
   }
   {
     set y x + 2*x;
     set x 2;
   }

   print z;
   print y;
}
